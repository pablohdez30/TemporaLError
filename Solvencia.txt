        private static bool ValidarCertificadoServidor(object sender, System.Security.Cryptography.X509Certificates.X509Certificate certificate, System.Security.Cryptography.X509Certificates.X509Chain chain, System.Net.Security.SslPolicyErrors sslPolicyErrors)
        {
            if (sslPolicyErrors == System.Net.Security.SslPolicyErrors.None)
            {
                return true; // Certificado válido
            }

            // Mostrar errores detectados
            //Console.WriteLine("Errores de certificado: " + sslPolicyErrors);

        public void DataTableToCSV_Dinamico(DataSet dataset, string strFilePath, string nombreProc, string pSesion)
        {


            DataTable tablaGarantias = dataset.Tables[0];
            //PHP: Metodo NO Dinamico //System.Text.StringBuilder DatosFichero = new System.Text.StringBuilder(30000);
            System.Text.StringBuilder DatosFichero2 = new System.Text.StringBuilder(30000);

            for (int i = 0; i < tablaGarantias.Columns.Count; i++)
            {
                string nombreColumna = tablaGarantias.Columns[i].ColumnName;
                DatosFichero2.Append("\"" + nombreColumna + "\"");

                if (i < tablaGarantias.Columns.Count - 1)
                    DatosFichero2.Append(",");
                else
                    DatosFichero2.Append(SLB.Utils.UtilsString.CrLf);
            }

            foreach (DataRow dr in tablaGarantias.Rows)
            {

                for (int i = 0; i < tablaGarantias.Columns.Count; i++)
                {
                    DataColumn column = tablaGarantias.Columns[i];
                    string valor = SIRBE.Utils.UtilsBBDD.DatosToObjeto<string>(dr[column]);

                    if (i < tablaGarantias.Columns.Count - 1) // Mientras no sea el útimo
                    {
                        DatosFichero2.Append("\"" + valor + "\",");
                    }
                    else
                    {
                        DatosFichero2.Append("\"" + valor + "\"");
                        DatosFichero2.Append(SLB.Utils.UtilsString.CrLf);
                    }
                }
            }
            try
            {
                using (StreamWriter sw = new StreamWriter(strFilePath, false))
                {
                    sw.Write(DatosFichero2); // Cuando podamos probar se usa DatosFichero2
                }
            }
            catch (Exception ex)
            {
                string mensajeFailed = "Fallo al guardar los datos de " + nombreProc + " desde bbdd al csv (DataTableToCSV_Dinamico),";
                mensajeFailed += " ERROR: " + ex.Message;
                SLB.ObjRemotos.TraceServidor.Trace.Show(pSesion, mensajeFailed, 5, "1");
                throw new Exception("<ERROR>GestorValidación en proceso DataTableToCSV_" + nombreProc + " :" + ex.Message + "<ERROR>");
            }

        }

        #region zipearCSV


        public void ComprimirCSV(string rutaCSV, string rutaZIP)
        {
            try
            {
                string nombreArchivoDentroZip = Path.GetFileName(rutaCSV);

                // Elimina el ZIP si ya existe
                if (File.Exists(rutaZIP))
                    File.Delete(rutaZIP);

                // Crea el ZIP y añade el archivo                   
                using (FileStream zipToOpen = new FileStream(rutaZIP, FileMode.Create))
                {
                    using (ZipArchive archive = new ZipArchive(zipToOpen, ZipArchiveMode.Create))
                    {
                        // Añade el archivo CSV dentro del ZIP con el nombre especificado
                        archive.CreateEntryFromFile(rutaCSV, nombreArchivoDentroZip);
                    }
                }
                if (File.Exists(rutaCSV))
                    File.Delete(rutaCSV);

            }
            catch (Exception ex)
            {
                string mensajeError = $"Error al comprimir el archivo CSV a ZIP: {ex.Message}";
                // Puedes registrar el error o lanzar una excepción
                throw new Exception(mensajeError);
            }
        }

        private string ObtenerToken()
        {
            string url = "";
            string token = string.Empty;

            try
            {
                HttpWebRequest request = (HttpWebRequest)WebRequest.Create(url);
                request.ServerCertificateValidationCallback = ValidarCertificadoServidor;
                request.Method = "GET";
                request.Accept = "application/json";

                using (HttpWebResponse response = (HttpWebResponse)request.GetResponse())
                {
                    using (StreamReader reader = new StreamReader(response.GetResponseStream()))
                    {
                        string responseBody = reader.ReadToEnd();
                        token = UtilsJson.ExtractValue(responseBody, "token");
                    }
                }
            }
            catch (WebException ex)
            {
                using (WebResponse response = ex.Response)
                {
                    if (response != null)
                    {
                        using (StreamReader reader = new StreamReader(response.GetResponseStream()))
                        {
                            string errorResponse = reader.ReadToEnd();
                            Console.WriteLine($"Error en la solicitud: {errorResponse}");
                        }
                    }
                    else
                    {
                        Console.WriteLine($"Error en la solicitud: {ex.Message}");
                    }
                }
            }
            return token;
        }

        public List<string> GetTablasConfiguracion(string urlBase, string token)
        {
            List<string> tablas = new List<string>();
            string requestUrl = $"{urlBase}/get-process-config";

            try
            {
                HttpWebRequest request = (HttpWebRequest)WebRequest.Create(requestUrl);
                request.ServerCertificateValidationCallback = ValidarCertificadoServidor;
                request.Method = "GET";
                request.Accept = "application/json";
                request.Headers.Add("Authorization", $"Bearer {token}");
                request.Timeout = 120000;

                using (HttpWebResponse response = (HttpWebResponse)request.GetResponse())
                {
                    using (StreamReader reader = new StreamReader(response.GetResponseStream()))
                    {
                        string responseBody = reader.ReadToEnd();

                        // Usar la función ExtractTables para obtener las tablas
                        tablas = UtilsJson.ExtractTables(responseBody, "credito", new string[] { "entrada_obligatoria", "entrada_opcional" });
                    }
                }
            }
            catch (WebException ex)
            {
                // Manejar errores de conexión
                if (ex.Response != null)
                {
                    using (var errorResponse = (HttpWebResponse)ex.Response)
                    using (var reader = new StreamReader(errorResponse.GetResponseStream()))
                    {
                        string errorBody = reader.ReadToEnd();
                        throw new Exception($"Error al obtener configuración: {errorBody}");
                    }
                }
                else
                {
                    throw new Exception("Error desconocido al conectar con el servidor.");
                }
            }

            return tablas;
        }

        /*
       09/12/2025 PHP:
       Los métodos SplitCsvLine() y SmartSplit() sirven para separar correctamente los valores de una línea CSV incluso cuando algunos campos contienen comas o comillas dentro del propio texto.
       A diferencia de `Split(',')`, estos métodos recorren la línea carácter a carácter y solo dividen cuando la coma está fuera de un bloque entrecomillado, respetando así el formato CSV real. 
       Gracias a esto, valores como `"J.P.MORG.AG,FFT"` se mantienen como un único campo y el número de columnas obtenidas coincide siempre con la estructura original del fichero.

       Los hemos añadido para cubrir el caso en la que el regex no realice bien el split
       */
        public static string[] SplitCsvLine(string line)
        {
            var result = new List<string>();
            var sb = new StringBuilder();
            bool inQuotes = false;

            for (int i = 0; i < line.Length; i++)
            {
                char c = line[i];

                if (c == '"')
                {
                    // Cambiamos el estado: dentro/fuera de comillas
                    inQuotes = !inQuotes;
                }
                else if (c == ',' && !inQuotes)
                {
                    // Coma separadora (solo si NO estamos dentro de comillas)
                    result.Add(sb.ToString());
                    sb.Clear();
                }
                else
                {
                    sb.Append(c);
                }
            }

            // Último campo
            result.Add(sb.ToString());

            return result.ToArray();
        }


        /*
         09/12/2025 PHP:
        El primer argumento s es el texto que quieres partir, el segundo, splitter, es el carácter que separa los tokens, el tercero, 
        quote es el carácter que indica el principio y el final de las comillas donde quieres evitar que se parta el texto, y el argumento final 
        includeQuotes define si los caracteres de comillas deben incluirse en la salida o omitirse. 
         */
        public static string[] SmartSplit(string s, char splitter, char quote, bool includeQuotes)
        {
            if (splitter == quote) throw new ArgumentException();
            List<string> tokens = new List<string>();
            StringBuilder sb = new StringBuilder();
            bool insideQuotes = false;
            for (int i = 0; i < s.Length; i++)
            {
                if (!insideQuotes && s[i] == splitter)
                {
                    tokens.Add(sb.ToString());
                    sb.Clear();
                    continue;
                }
                if (s[i] == quote)
                {
                    insideQuotes = !insideQuotes;
                    if (includeQuotes)
                        sb.Append(quote);
                    continue;
                }
                sb.Append(s[i]);
            }
            if (sb.Length > 0)
                tokens.Add(sb.ToString());
            return tokens.ToArray();
        }

                    reader.ReadLine();
                    numLineasCSV++;

            modoBulk = numLineasCSV >= 50000; //Elegimos el modo bulk o el script;

        }

        private string convierteNUMERIC_VacioADBNull(string pValor, int fila)
        {
            if (string.IsNullOrWhiteSpace(pValor))
                return "NULL";

            if (decimal.TryParse(pValor, System.Globalization.NumberStyles.Any,
                                 System.Globalization.CultureInfo.InvariantCulture, out decimal resultadoDecimal))
            {
                return resultadoDecimal.ToString(System.Globalization.CultureInfo.InvariantCulture);
            }
            else
            {
                throw new Exception($"Numeric no válido en la fila: {fila}");
            }
        }

        private string convierteDATETIME_VacioADBNull(string pValor, int fila)
        {
            if (string.IsNullOrWhiteSpace(pValor))
                return "NULL";

            DateTime fecha;
            if (DateTime.TryParseExact(pValor, "yyyy-MM-dd", System.Globalization.CultureInfo.InvariantCulture, System.Globalization.DateTimeStyles.None, out fecha))
            {
                return $"'{fecha:yyyy-MM-dd}'";
            }
            else
            {
                throw new Exception($"DateTime no válido en la fila: {fila}");
            }
        }

        public byte[] recuperaExcelStream(string pExcel)
        {
            try
            {
                string pathPlantillaExcel = Path.GetDirectoryName(Assembly.GetExecutingAssembly().Location) + pExcel;

                using (FileStream fsSource = new FileStream(pathPlantillaExcel, FileMode.Open, FileAccess.Read))
                {
                    byte[] bytes = new byte[fsSource.Length];
                    int numBytesToRead = (int)fsSource.Length;
                    int numBytesRead = 0;
                    while (numBytesToRead > 0)
                    {
                        int n = fsSource.Read(bytes, numBytesRead, numBytesToRead);

                        // Salimos al llegar al final del fichero.
                        if (n == 0)
                            break;

                        numBytesRead += n;
                        numBytesToRead -= n;
                    }

                    return bytes;
                }

            }
            catch (Exception ex)
            {
                throw new Exception("<ERROR>Error en file stream libro Excel:" + ex.Message + "</ERROR>");
            }
        }













        private bool ValidarUnFicheroHastaFin(string pSesion, string fileBase, string fileId, string idioma, string mes, string año, string url, string token, TimeSpan intervaloPolling, TimeSpan timeoutPorFichero)
        {
            var sw = System.Diagnostics.Stopwatch.StartNew();

            int intentos = 0;

            TraceServidor.Trace.Show(pSesion, $"[SOLV][VALID] INICIO fileBase={fileBase} fileId={fileId} ", 4, idioma);

            while (true)
            {
                if (sw.Elapsed > timeoutPorFichero)
                {
                    SLB.ObjRemotos.TraceServidor.Trace.ShowError(pSesion, $"[SOLV][VALID] TIMEOUT fileBase={fileBase} fileId={fileId} min={sw.Elapsed.TotalMinutes:F1}", idioma);
                    return false;
                }

                string stateUrl = $"{url}/load/{fileBase}/{fileId}/state?deletePod=True&month={mes}&year={año}";

                try
                {
                    HttpWebRequest request = (HttpWebRequest)WebRequest.Create(stateUrl);
                    request.ServerCertificateValidationCallback = ValidarCertificadoServidor;
                    request.Method = "GET";
                    request.Accept = "application/json";
                    request.Timeout = 360000;
                    request.Headers.Add("Authorization", $"Bearer {token}");

                    using (HttpWebResponse response = (HttpWebResponse)request.GetResponse())
                    using (StreamReader reader = new StreamReader(response.GetResponseStream()))
                    {
                        string responseBody = reader.ReadToEnd();
                        string res = UtilsJson.EstadoFinalizacion(responseBody);

                        if (res == "Pending" || res == "Running")
                        {
                            // continuar
                        }
                        else if (res == "Succeeded")
                        {
                            TraceServidor.Trace.Show(pSesion, $"[SOLV][VALID] OK fileBase={fileBase} fileId={fileId} min={sw.Elapsed.TotalMinutes:F1}", 4, idioma);
                            return true;
                        }
                        else
                        {
                            SLB.ObjRemotos.TraceServidor.Trace.ShowError(pSesion, $"[SOLV][VALID] FAIL fileBase={fileBase} fileId={fileId} estado={res} min={sw.Elapsed.TotalMinutes:F1} ", idioma);
                            return false;
                        }
                    }
                }
                catch (WebException ex)
                {
                    string body = "";
                    int code = 0;
                    if (ex.Response is HttpWebResponse errorResponse)
                    {
                        code = (int)errorResponse.StatusCode;
                        using (var reader = new StreamReader(errorResponse.GetResponseStream()))
                            body = reader.ReadToEnd();
                    }

                    SLB.ObjRemotos.TraceServidor.Trace.ShowError(
                        pSesion, $"[SOLV][VALID] WebException fileBase={fileBase} fileId={fileId} HTTP={code} min={sw.Elapsed.TotalMinutes:F1} ERROR={ex.Message} BODY={body}", idioma);
                    return false;
                }

                System.Threading.Thread.Sleep(intervaloPolling);
            }
        }
        private string UploadFileToSolvencia(string fileBase, string filePath, string idSesion, string idioma, string añoProceso, string mesProceso, string baseUrl, string rutaTemporalSolvencia, string token, bool useAuthHeader, bool validations, int timeoutMs = 120000)
        {
            string validationStr = validations ? "True" : "False";
            string requestUrl = $"{baseUrl}/load/{fileBase}?validations={validationStr}&month={mesProceso}&year={añoProceso}";

            // Paths de debug 
            string responseBodyPath = System.IO.Path.Combine(rutaTemporalSolvencia, $"response_body_{fileBase}.txt");
            string responseCodePath = System.IO.Path.Combine(rutaTemporalSolvencia, $"response_code_{fileBase}.txt");

            TraceServidor.Trace.Show(idSesion, $"[SOLV][UPLOAD] INICIO fileBase={fileBase} file={Path.GetFileName(filePath)}", 4, idioma);

            try
            {
                string boundary = "----Boundary" + DateTime.UtcNow.Ticks; // boundary único 
                string headerTemplate =
                    $"--{boundary}\r\n" +
                    $"Content-Disposition: form-data; name=\"uploaded_file\"; filename=\"{Path.GetFileName(filePath)}\"\r\n" +
                    $"Content-Type: application/octet-stream\r\n\r\n";
                string footer = $"\r\n--{boundary}--\r\n";

                HttpWebRequest request = (HttpWebRequest)WebRequest.Create(requestUrl);
                request.ServerCertificateValidationCallback = ValidarCertificadoServidor;
                request.Method = "POST";
                request.Accept = "application/json";
                request.ContentType = $"multipart/form-data; boundary={boundary}";
                request.Timeout = timeoutMs;

                if (useAuthHeader && !string.IsNullOrEmpty(token))
                    request.Headers.Add("Authorization", $"Bearer {token}");

                byte[] fileData = File.ReadAllBytes(filePath);
                byte[] headerBytes = Encoding.UTF8.GetBytes(headerTemplate);
                byte[] footerBytes = Encoding.UTF8.GetBytes(footer);

                request.ContentLength = headerBytes.Length + fileData.Length + footerBytes.Length;

                using (Stream requestStream = request.GetRequestStream())
                {
                    requestStream.Write(headerBytes, 0, headerBytes.Length);
                    requestStream.Write(fileData, 0, fileData.Length);
                    requestStream.Write(footerBytes, 0, footerBytes.Length);
                }

                using (HttpWebResponse response = (HttpWebResponse)request.GetResponse())
                using (StreamReader reader = new StreamReader(response.GetResponseStream()))
                {
                    int httpCode = (int)response.StatusCode;
                    string responseBody = reader.ReadToEnd();

                    File.WriteAllText(responseCodePath, httpCode.ToString());
                    File.WriteAllText(responseBodyPath, responseBody);

                    if (httpCode > 210)
                        throw new Exception($"HTTP {httpCode} en upload {fileBase}. ");

                    string fileId = UtilsJson.ExtractValue(responseBody, "fileID");
                    if (string.IsNullOrEmpty(fileId))
                        throw new Exception($"No se obtuvo fileID en upload {fileBase}. ");


                    // NUEVO: execution_id para que puedan seguir mejor el rastro si falla
                    string executionId = UtilsJson.ExtractValue(responseBody, "execution_id");
                    if (!string.IsNullOrEmpty(executionId))
                    {
                        TraceServidor.Trace.Show(idSesion, $"[SOLV][UPLOAD] OK fileBase={fileBase} fileId={fileId} execution_id={executionId} ", 4, idioma);
                    }
                    else
                    {
                        throw new Exception($"No se obtuvo execution_id en upload {fileBase}. ");
                    }

                    // Limpieza de debug 
                    SLB.ObjRemotos.Comun.Compresor.SafeDelete(responseBodyPath);
                    SLB.ObjRemotos.Comun.Compresor.SafeDelete(responseCodePath);


                    if (validations)
                        System.Threading.Thread.Sleep(10000); // 10 seg

                    return fileId;
                }
            }
            catch (WebException ex)
            {
                string errorBody = "";
                int httpCode = 0;

                if (ex.Response is HttpWebResponse errorResponse)
                {
                    httpCode = (int)errorResponse.StatusCode;
                    using (var reader = new StreamReader(errorResponse.GetResponseStream()))
                        errorBody = reader.ReadToEnd();
                }

                File.WriteAllText(responseCodePath, httpCode.ToString());
                File.WriteAllText(responseBodyPath, errorBody);

                string msg = $"UPLOAD FALLIDO {fileBase}. HTTP={httpCode}. ERROR={ex.Message}. BODY={errorBody}";
                SLB.ObjRemotos.TraceServidor.Trace.ShowError(idSesion, msg, idioma);

                throw; // importantísimo: aborta
            }
            catch (Exception ex)
            {
                SLB.ObjRemotos.TraceServidor.Trace.ShowError(idSesion, $"UPLOAD FALLIDO {fileBase}. ERROR={ex.Message}", idioma);
                throw;
            }
        }



        public string ProcesarSecuencialSubidaYValidacion(string pSesion, string idioma, string mes, string año, string urlMotorSolvencia, string rutaTemporalSolvencia, string token, string enviarZipeadosLosFicheros, List<FicheroSolvencia> ficheros, List<string> tablasNecesarias, string pCadenaEntrada, bool tieneContenidoContabilidad, bool tieneContenidoMapeo)
        {
            TimeSpan pollCada20s = TimeSpan.FromSeconds(20);
            TimeSpan timeoutPorFichero20m = TimeSpan.FromMinutes(20);

            string PickPath(string csv, string zip)
                => (enviarZipeadosLosFicheros == "1") ? zip : csv;

            TraceServidor.Trace.Show(pSesion, $"[SOLV] INICIO Subida+Validacion. Mes={mes} Año={año}", 5, idioma);

            bool validations;
            if (pCadenaEntrada == "SOLV_CREDITO")
            {
                validations = true;
            }
            else //SOLV_OPERACIONAL
            {
                validations = false;
            }

            try
            {
                string lista = string.Join(", ",
                    ficheros
                        .Where(x => tablasNecesarias.Contains(x.FileBase))
                        .Select(x => x.FileBase));
                TraceServidor.Trace.Show(pSesion, $"[SOLV][SEQ] Se van a procesar: {lista}", 4, idioma);
            }
            catch { /* no quiero romper por un log */ }

            // ===============================
            // SUBIDA + VALIDACIÓN SECUENCIAL
            // ===============================
            foreach (var f in ficheros)
            {
                if (!tablasNecesarias.Contains(f.FileBase))
                    continue;

                string path = PickPath(f.CsvPath, f.ZipPath);

                string fileId = null;
                var swUpload = System.Diagnostics.Stopwatch.StartNew();
                try
                {
                    TraceServidor.Trace.Show(pSesion, $"[SOLV][{f.FileBase}] UPLOAD -> POST /load/{f.FileBase}", 4, idioma);

                    fileId = UploadFileToSolvencia(
                        f.FileBase,
                        path,
                        pSesion,
                        idioma,
                        año,
                        mes,
                        urlMotorSolvencia,
                        rutaTemporalSolvencia,
                        token,
                        f.UseAuthHeader,
                        validations
                    );

                    f.SetFileId(fileId);

                    TraceServidor.Trace.Show(pSesion,
                        $"[SOLV][{f.FileBase}] UPLOAD OK. fileId={fileId} TiempoMs={swUpload.ElapsedMilliseconds}", 5, idioma);
                }
                catch (Exception ex)
                {
                    TraceServidor.Trace.ShowError(pSesion,
                        $"[SOLV][{f.FileBase}] UPLOAD FAIL. TiempoMs={swUpload.ElapsedMilliseconds}. ERROR={ex.Message}", idioma);
                    return "failed";
                }

                // 2) Validación (polling)
                var swVal = System.Diagnostics.Stopwatch.StartNew();
                TraceServidor.Trace.Show(pSesion,
                    $"[SOLV][{f.FileBase}] VALIDACION INICIO. PollCada={pollCada20s.TotalSeconds}s Timeout={timeoutPorFichero20m.TotalMinutes}m fileId={fileId}", 5, idioma);

                bool ok;
                try
                {
                    ok = ValidarUnFicheroHastaFin(
                        pSesion,
                        f.FileBase,
                        fileId,
                        idioma,
                        mes,
                        año,
                        urlMotorSolvencia,
                        token,
                        pollCada20s,
                        timeoutPorFichero20m
                    );
                }
                catch (Exception ex)
                {
                    TraceServidor.Trace.ShowError(pSesion,
                        $"[SOLV][{f.FileBase}] VALIDACION EXCEPCION. TiempoMs={swVal.ElapsedMilliseconds}. ERROR={ex.Message}", idioma);
                    return "failed";
                }

                if (!ok)
                {
                    TraceServidor.Trace.ShowError(pSesion, $"[SOLV][{f.FileBase}] VALIDACION FAIL. fileId={fileId} TiempoMs={swVal.ElapsedMilliseconds}", idioma);
                    return "failed";
                }

                TraceServidor.Trace.Show(pSesion,
                    $"[SOLV][{f.FileBase}] VALIDACION OK. fileId={fileId} TiempoMs={swVal.ElapsedMilliseconds}", 5, idioma);

            }

            // ===============================
            // EJECUCIÓN GLOBAL
            // ===============================

            TraceServidor.Trace.Show(pSesion, $"[SOLV][EXEC] INICIO POST + Polling. ", 5, idioma);

            var swExec = System.Diagnostics.Stopwatch.StartNew();
            string processId;
            try
            {
                processId = LanzarEjecucionYEsperar(
                    pSesion,
                    idioma,
                    mes,
                    año,
                    urlMotorSolvencia,
                    rutaTemporalSolvencia,
                    token,
                    pCadenaEntrada,
                    tieneContenidoContabilidad,
                    tieneContenidoMapeo
                );
            }
            catch (Exception ex)
            {
                TraceServidor.Trace.ShowError(pSesion,
                    $"[SOLV][EXEC] ERROR en POST + Polling. TiempoMs={swExec.ElapsedMilliseconds}. ERROR={ex.Message}", idioma);
                return "failed";
            }

            TraceServidor.Trace.Show(pSesion, $"[SOLV][EXEC] FIN de POST + Polling. TiempoMs={swExec.ElapsedMilliseconds}", 5, idioma);

            TraceServidor.Trace.Show(pSesion, $"[SOLV][SEQ] FIN Subida+Validacion. Resultado={processId}", 5, idioma);
            return processId;
        }


        public string LanzarEjecucionYEsperar(string pSesion, string idioma, string mes, string año,
            string url, string rutaTemporalSolvencia, string token,
            string cadenaEntrada, bool tieneContenidoContabilidad, bool tieneContenidoMapeo)
        {
            string runVersion = "CRR3";
            string lastLoad = "True";
            string executionUrl = "";

            if (cadenaEntrada == "SOLV_OPERACIONAL")
            {
                executionUrl =
                    $"{url}/execution/operational-risk" +
                    $"?month={mes}&year={año}&lastLoad={lastLoad}";
            }
            else //Cadena Credito (SOLV_CREDITO)
            {
                string auxContabilidad = tieneContenidoContabilidad ? "True" : "False";
                string auxMapeoCCNS = tieneContenidoMapeo ? "True" : "False";

                executionUrl =
                    $"{url}/execution/counterpart-credit-risk" +
                    $"?MapeoCCNSEpigrafe_RC={auxMapeoCCNS}" +
                    $"&Contabilidad_RC={auxContabilidad}" +
                    $"&runVersion={runVersion}&month={mes}&year={año}&lastLoad={lastLoad}";
            }


            string processId;
            string añoPost = año;
            string mesPost = mes;

            string executionResponsePath = Path.Combine(rutaTemporalSolvencia, "execution_response.json");
            string executionHttpCodePath = Path.Combine(rutaTemporalSolvencia, "execution_http_code.txt");
            string stateResponseFile = Path.Combine(rutaTemporalSolvencia, "state_response.json");

            try
            {
                HttpWebRequest request = (HttpWebRequest)WebRequest.Create(executionUrl);
                request.ServerCertificateValidationCallback = ValidarCertificadoServidor;
                request.Method = "POST";
                request.ContentType = "application/json";
                request.Accept = "application/json";
                request.Headers.Add("Authorization", $"Bearer {token}");

                using (StreamWriter writer = new StreamWriter(request.GetRequestStream()))
                    writer.Write("{}");

                using (HttpWebResponse postResponse = (HttpWebResponse)request.GetResponse())
                using (StreamReader reader = new StreamReader(postResponse.GetResponseStream()))
                {
                    int httpCode = (int)postResponse.StatusCode;
                    File.WriteAllText(executionHttpCodePath, httpCode.ToString());

                    string postResponseBody = reader.ReadToEnd();
                    File.WriteAllText(executionResponsePath, postResponseBody);

                    if (httpCode != 202)
                        return "failed";

                    processId = UtilsJson.ExtractValue(postResponseBody, "id");
                    (añoPost, mesPost) = UtilsJson.ExtractYearAndMonthFromJson(postResponseBody, "date");

                    if (string.IsNullOrEmpty(processId))
                        return "failed";
                }
            }
            catch (WebException ex)
            {
                int httpCode = 0;
                string body = "";

                if (ex.Response is HttpWebResponse errorResponse)
                {
                    httpCode = (int)errorResponse.StatusCode;
                    using (var reader = new StreamReader(errorResponse.GetResponseStream()))
                        body = reader.ReadToEnd();
                }

                File.WriteAllText(executionHttpCodePath, httpCode.ToString());
                File.WriteAllText(executionResponsePath, body);

                SLB.ObjRemotos.TraceServidor.Trace.ShowError(
                    pSesion,
                    $"EJECUCION FALLIDA. HTTP={httpCode}. ERROR={ex.Message}. BODY={body}",
                    idioma
                );

                return "failed";
            }

            string deletePodBool = "True";
            int retryCount = 0;
            int maxRetries = 300;                // parecido al BAT
            int sleepMs = 150000;                // 2,5 min
            int logEvery = 5;                    // log cada 5 intentos (ajusta)

            bool loggedInitialState = false;     // Es una flag que sirve para saber si ya hemos escrito en el log el estado inicial del proceso.

            while (true)
            {
                retryCount++;
                string stateUrl = $"{url}/execution/{processId}/state?month={mesPost}&year={añoPost}&deletePod={deletePodBool}";

                try
                {
                    HttpWebRequest request = (HttpWebRequest)WebRequest.Create(stateUrl);
                    request.ServerCertificateValidationCallback = ValidarCertificadoServidor;
                    request.Method = "GET";
                    request.Accept = "application/json";
                    request.Headers.Add("Authorization", $"Bearer {token}");

                    using (HttpWebResponse pollResponse = (HttpWebResponse)request.GetResponse())
                    using (StreamReader reader = new StreamReader(pollResponse.GetResponseStream()))
                    {
                        int httpCode = (int)pollResponse.StatusCode;
                        string pollBody = reader.ReadToEnd();

                        File.WriteAllText(executionHttpCodePath, httpCode.ToString());
                        File.WriteAllText(stateResponseFile, pollBody ?? "");

                        if (!loggedInitialState)
                        {
                            // 1) estado inicial (equivalente al BAT)
                            SLB.ObjRemotos.TraceServidor.Trace.Show(
                                pSesion,
                                $"[Polling] Estado inicial. processId={processId}, HTTP={httpCode}, month={mesPost}, year={añoPost}. Body={pollBody}", 5,
                                idioma
                            );
                            loggedInitialState = true;
                        }
                        else if (retryCount % logEvery == 0)
                        {
                            // 2) log de progreso cada X (5) intentos
                            string estadoTmp = UtilsJson.EstadoFinalizacion(pollBody) ?? "Unknown";
                            SLB.ObjRemotos.TraceServidor.Trace.Show(
                                pSesion,
                                $"[Polling] Reintento {retryCount}/{maxRetries}. Estado={estadoTmp}. HTTP={httpCode}.", 5,
                                idioma
                            );
                        }

                        // 3) manejo normal
                        string res = UtilsJson.EstadoFinalizacion(pollBody);

                        if (res == "Succeeded")
                        {
                            SLB.ObjRemotos.TraceServidor.Trace.Show(
                                pSesion,
                                $"[Polling] Proceso finalizado con Succeeded. processId={processId}. Intentos={retryCount}.", 5,
                                idioma
                            );

                            SLB.ObjRemotos.Comun.Compresor.SafeDelete(stateResponseFile);
                            SLB.ObjRemotos.Comun.Compresor.SafeDelete(executionResponsePath);
                            SLB.ObjRemotos.Comun.Compresor.SafeDelete(executionHttpCodePath);
                            return processId;
                        }
                        else if (res == "Failed")
                        {
                            SLB.ObjRemotos.TraceServidor.Trace.ShowError(
                                pSesion,
                                $"[Polling] Proceso finalizado con Failed. processId={processId}. Intentos={retryCount}. Body={pollBody}",
                                idioma
                            );

                            SLB.ObjRemotos.Comun.Compresor.SafeDelete(stateResponseFile);
                            SLB.ObjRemotos.Comun.Compresor.SafeDelete(executionResponsePath);
                            SLB.ObjRemotos.Comun.Compresor.SafeDelete(executionHttpCodePath);
                            return "failed";
                        }

                        // 4) tope de reintentos 
                        if (retryCount >= maxRetries)
                        {
                            SLB.ObjRemotos.TraceServidor.Trace.ShowError(
                                pSesion,
                                $"[Polling] No alcanzó Succeeded tras {maxRetries} intentos. Último HTTP={httpCode}. Último Body={pollBody}",
                                idioma
                            );

                            SLB.ObjRemotos.Comun.Compresor.SafeDelete(stateResponseFile);
                            SLB.ObjRemotos.Comun.Compresor.SafeDelete(executionResponsePath);
                            SLB.ObjRemotos.Comun.Compresor.SafeDelete(executionHttpCodePath);
                            return "failed";
                        }
                    }
                }
                catch (WebException ex)
                {
                    // Log con HTTP 
                    int httpCode = 0;
                    string body = "";

                    if (ex.Response is HttpWebResponse errorResponse)
                    {
                        httpCode = (int)errorResponse.StatusCode;
                        using (var r = new StreamReader(errorResponse.GetResponseStream()))
                            body = r.ReadToEnd();
                    }

                    SLB.ObjRemotos.TraceServidor.Trace.ShowError(
                        pSesion,
                        $"[Polling] Error HTTP en intento {retryCount}/{maxRetries}. HTTP={httpCode}. MSG={ex.Message}. Body={body}",
                        idioma
                    );

                    return "failed";
                }
                catch (Exception ex)
                {
                    System.Reflection.MethodBase metodo = System.Reflection.MethodBase.GetCurrentMethod();
                    string llamadaMetodo = metodo.DeclaringType.FullName + "." + metodo.Name;
                    string mensajeSimple = $"Error en {llamadaMetodo}:{ex.Message}";

                    SLB.ObjRemotos.TraceServidor.Trace.ShowError(pSesion, mensajeSimple, idioma);
                    throw new SIRBE.Servicios.Comun.ServicioException(mensajeSimple, ex);
                }

                System.Threading.Thread.Sleep(sleepMs);
            }


        }
